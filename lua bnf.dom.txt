*integer		::= {0-9} [{0-9}]*
?float		::= [integer] . [integer] 
*number		::= integer | float
*boolean	::= 'true' | 'false'
*primary         ::= 'nil' | boolean | number | ident | {'(' rvalue ')'}
*chain           ::= primary [{ '*' | '/' } primary]
*rvalue          ::= chain [{ '+' | '-' } chain]
unop            ::= [{'-' | '+'}] rvalue
*ident           ::= {a-zA-Z} [{a-zA-Z0-9}]*
*assign      ::= ['local'] ident ['=' [ident '=']* rvalue]*
